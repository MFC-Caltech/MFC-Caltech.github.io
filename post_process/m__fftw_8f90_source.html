71  SUBROUTINE s_initialize_fftw_module() ! ----------------------------------
72 
73  ! Size of input array going into DFT
74  real_size = p+1
75  ! Size of output array coming out of DFT
76  cmplx_size = (p+1)/2+1
77 
78  ! Allocate input and output DFT data sizes
79  fftw_real_data = fftw_alloc_real(int( real_size, c_size_t))
80  fftw_cmplx_data = fftw_alloc_complex(int(cmplx_size, c_size_t))
81  fftw_fltr_cmplx_data = fftw_alloc_complex(int(cmplx_size, c_size_t))
82  ! Associate input and output data pointers with allocated memory
83  CALL c_f_pointer(fftw_real_data , data_real , [ real_size])
84  CALL c_f_pointer(fftw_cmplx_data, data_cmplx, [cmplx_size])
85  CALL c_f_pointer(fftw_fltr_cmplx_data, data_fltr_cmplx, [cmplx_size])
86 
87  ! Generate plans for forward and backward DFTs
88  fwd_plan = fftw_plan_dft_r2c_1d(real_size, data_real , data_cmplx, fftw_estimate)
89  bwd_plan = fftw_plan_dft_c2r_1d(real_size, data_fltr_cmplx, data_real , fftw_estimate)

91  END SUBROUTINE s_initialize_fftw_module ! ------------------------------

101  SUBROUTINE s_apply_fourier_decomposition(q_sf,i) ! -----------------------
102 
103  ! Variable to be Fourier decomposed
104  REAL(KIND(0d0)), &
105  DIMENSION( -offset_x%beg : m+offset_x%end , & -offset_y%beg : n+offset_y%end , & -offset_z%beg : p+offset_z%end ), &
106  INTENT(INOUT) :: q_sf
107 
108  INTEGER, INTENT(IN) :: i
109 
110  INTEGER :: j,k
111 
112  DO j = -offset_x%beg, m+offset_x%end
113  DO k = -offset_y%beg, n+offset_y%end
114  data_fltr_cmplx(:) = (0d0,0d0)
115  data_real(1:p+1) = q_sf(j,k,0:p)
116  CALL fftw_execute_dft_r2c(fwd_plan, data_real, data_cmplx)
117  data_fltr_cmplx(i) = data_cmplx(i)
118  CALL fftw_execute_dft_c2r(bwd_plan, data_fltr_cmplx, data_real)
119  data_real(:) = data_real(:)/REAL(real_size,kind(0d0))
120  q_sf(j,k,0:p) = data_real(1:p+1)
121  END DO
122  END DO
123 
124  ! Populate offset regions given that domain is azimuthally periodic
125  DO j = -offset_z%beg, -1
126  q_sf(:,:,j) = q_sf(:,:,(p+1)+j)
127  END DO
128  DO j = 1, offset_z%end
129  q_sf(:,:,p+j) = q_sf(:,:,j-1)
130  END DO
131 
132  END SUBROUTINE s_apply_fourier_decomposition ! -------------------------
