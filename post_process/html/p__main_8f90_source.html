617  dft_q_sf(:,:,:) = q_sf(:,:,:)
618  DO j = fourier_modes%beg, fourier_modes%end

620  CALL s_apply_fourier_decomposition(q_sf,j)
621  WRITE(varname, '(A,I0,A,I0)') 'omega', i, '_', j
622  CALL s_write_variable_to_formatted_database_file(varname,t_step)
623  END DO

643  DO j = fourier_modes%beg, fourier_modes%end
644  q_sf(:,:,:) = dft_q_sf(:,:,:)
645  CALL s_apply_fourier_decomposition(q_sf,j)

648  END DO

652  END IF

655 
656  END IF

658

663  -offset_x%beg : m + offset_x%end, &
664  -offset_y%beg : n + offset_y%end, &
665  -offset_z%beg : p + offset_z%end )
666 
667  IF (fourier_decomp) THEN

669  DO j = fourier_modes%beg, fourier_modes%end
670  q_sf(:,:,:) = dft_q_sf(:,:,:)
671  CALL s_apply_fourier_decomposition(q_sf,j)
672  WRITE(varname, '(A,I0,A,I0)') 'alpha', i-e_idx, '_', j
673  CALL s_write_variable_to_formatted_database_file(varname,t_step)
674  END DO
675  ELSE
676  WRITE(varname, '(A,I0)') 'alpha', i-e_idx
677  CALL s_write_variable_to_formatted_database_file(varname,t_step)
678  END IF
679  varname(:) = ' '
680  END DO
681  END IF
682

684  IF (bubbles) THEN

686  DO i = 1,nb
687  q_sf = q_cons_vf(bub_idx%rs(i))%sf( &
688  -offset_x%beg : m + offset_x%end, &
689  -offset_y%beg : n + offset_y%end, &
690  -offset_z%beg : p + offset_z%end )
691  WRITE(varname, '(A,I3.3)') 'nR', i
692  CALL s_write_variable_to_formatted_database_file(varname,t_step)
693  varname(:) = ' '
694  END DO
695

697  DO i = 1,nb
698  q_sf = q_cons_vf(bub_idx%vs(i))%sf( &
699  -offset_x%beg : m + offset_x%end, &
700  -offset_y%beg : n + offset_y%end, &
701  -offset_z%beg : p + offset_z%end )
702  WRITE(varname, '(A,I3.3)') 'nV', i
703  CALL s_write_variable_to_formatted_database_file(varname,t_step)

708  DO i = 1,nb

710  -offset_x%beg : m + offset_x%end, &
711  -offset_y%beg : n + offset_y%end, &
712  -offset_z%beg : p + offset_z%end )
713  WRITE(varname, '(A,I3.3)') 'nP', i
714  CALL s_write_variable_to_formatted_database_file(varname,t_step)
715  varname(:) = ' '
716  END DO

719  DO i = 1,nb
720  q_sf = q_cons_vf(bub_idx%ms(i))%sf( &
721  -offset_x%beg : m + offset_x%end, &
722  -offset_y%beg : n + offset_y%end, &
723  -offset_z%beg : p + offset_z%end )
724  WRITE(varname, '(A,I3.3)') 'nM', i
725  CALL s_write_variable_to_formatted_database_file(varname,t_step)
726  varname(:) = ' '
727  END DO
728  END IF
729  END IF
730

732  CALL s_close_formatted_database_file()
733 
734  ! Modifies the time-step iterator so that it may reach the final time-
735  ! step to be post-processed, in the case that this one is not originally
736  ! attainable through constant incrementation from the first time-step.
737  ! This modification is performed upon reaching the final time-step. In
738  ! case that it is not needed, the post-processor is done and may exit.
739  IF((t_step_stop-t_step) < t_step_save .AND. t_step_stop /= t_step) THEN
740  t_step = t_step_stop - t_step_save
741  ELSEIF(t_step == t_step_stop) THEN
742  EXIT
743  END IF
744 
745  ! Incrementing time-step iterator to next time-step to be post-processed
746  t_step = t_step + t_step_save
747 
748  END DO
749  ! END: Time-Marching Loop ==================================================
750 
751  CLOSE(11)
752 
753  ! Disassociate pointers for serial and parallel I/O
754  s_read_data_files => null()
755

758  CALL s_finalize_derived_variables_module()
759  CALL s_finalize_data_input_module()

762  IF(num_procs > 1) CALL s_finalize_mpi_proxy_module()

767  CALL s_mpi_finalize()
