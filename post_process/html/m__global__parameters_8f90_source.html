381  SUBROUTINE s_initialize_global_parameters_module() ! ----------------------
382 
383  INTEGER :: i, fac
384 
385  ! Setting m_root equal to m in the case of a 1D serial simulation
386  IF(num_procs == 1 .AND. n == 0) m_root = m
387 
388 
389  ! Gamma/Pi_inf Model ===============================================
390  IF(model_eqns == 1) THEN
391 
392  ! Setting number of fluids
393  num_fluids = 1
394 
395  ! Annotating structure of the state and flux vectors belonging
396  ! to the system of equations defined by the selected number of
397  ! spatial dimensions and the gamma/pi_inf model
398  cont_idx%beg = 1
399  cont_idx%end = cont_idx%beg
400  mom_idx%beg = cont_idx%end + 1
401  mom_idx%end = cont_idx%end + num_dims
402  e_idx = mom_idx%end + 1
403  adv_idx%beg = e_idx + 1
404  adv_idx%end = adv_idx%beg + 1
405  gamma_idx = adv_idx%beg
406  pi_inf_idx = adv_idx%end
407  sys_size = adv_idx%end
408 
409  ! ==================================================================
410 
411 
412  ! Volume Fraction Model (5-equation model) =========================
413  ELSE IF(model_eqns == 2) THEN
414 
415  ! Annotating structure of the state and flux vectors belonging
416  ! to the system of equations defined by the selected number of
417  ! spatial dimensions and the volume fraction model
418  cont_idx%beg = 1
419  cont_idx%end = num_fluids
420  mom_idx%beg = cont_idx%end + 1
421  mom_idx%end = cont_idx%end + num_dims
422  e_idx = mom_idx%end + 1
423  adv_idx%beg = e_idx + 1
424  adv_idx%end = e_idx + num_fluids
425 
426  IF( (adv_alphan .NEQV. .true.) .AND. &
427  (num_fluids > 1)) adv_idx%end = adv_idx%end - 1
428 
429  sys_size = adv_idx%end
430 
431  IF (bubbles) THEN
432  alf_idx = adv_idx%end
433  ELSE
434  alf_idx = 0
435  END IF
436 
437  IF (bubbles) THEN
438  bub_idx%beg = sys_size+1
439  bub_idx%end = sys_size+2*nb
440  IF (polytropic .NEQV. .true.) THEN
441  bub_idx%end = sys_size+4*nb
442  END IF
443  sys_size = bub_idx%end
444 
445  ALLOCATE( bub_idx%rs(nb), bub_idx%vs(nb) )
446  ALLOCATE( bub_idx%ps(nb), bub_idx%ms(nb) )
447  ALLOCATE( weight(nb),r0(nb),v0(nb) )
448 
449  DO i = 1, nb
450  IF (polytropic .NEQV. .true.) THEN
451  fac = 4
452  ELSE
453  fac = 2
454  END IF
455 
456  bub_idx%rs(i) = bub_idx%beg+(i-1)*fac
457  bub_idx%vs(i) = bub_idx%rs(i)+1
458 
459  IF (polytropic .NEQV. .true.) THEN
460  bub_idx%ps(i) = bub_idx%vs(i)+1
461  bub_idx%ms(i) = bub_idx%ps(i)+1
462  END IF
463  END DO
464 
465  IF (nb == 1) THEN
466  weight(:) = 1d0
467  r0(:) = 1d0
468  v0(:) = 0d0
469  ELSE IF (nb > 1) THEN
470  CALL s_simpson(nb)
471  v0(:) = 0d0
472  ELSE
473  stop 'Invalid value of nb'
474  END IF
475 
476  IF (polytropic .NEQV. .true.) THEN
477  CALL s_initialize_nonpoly
478  ELSE
479  rhoref = 1.d0
480  pref = 1.d0
481  END IF
482  END IF
483 
484  ! ==================================================================
485 
486  ! Volume Fraction Model (6-equation model) =========================
487  ELSE IF(model_eqns == 3) THEN
488 
489  ! Annotating structure of the state and flux vectors belonging
490  ! to the system of equations defined by the selected number of
491  ! spatial dimensions and the volume fraction model
492  cont_idx%beg = 1
493  cont_idx%end = num_fluids
494  mom_idx%beg = cont_idx%end + 1
495  mom_idx%end = cont_idx%end + num_dims
496  e_idx = mom_idx%end + 1
497  adv_idx%beg = e_idx + 1
498  adv_idx%end = e_idx + num_fluids
499  IF(adv_alphan .NEQV. .true.) adv_idx%end = adv_idx%end - 1
500  internalenergies_idx%beg = adv_idx%end + 1
501  internalenergies_idx%end = adv_idx%end + num_fluids
502  sys_size = internalenergies_idx%end
503 
504  ELSE IF(model_eqns == 4) THEN
505  cont_idx%beg = 1 ! one continuity equation
506  cont_idx%end = 1 !num_fluids
507  mom_idx%beg = cont_idx%end + 1 ! one momentum equation in each
508  mom_idx%end = cont_idx%end + num_dims
509  e_idx = mom_idx%end + 1 ! one energy equation
510  adv_idx%beg = e_idx + 1
511  adv_idx%end = adv_idx%beg !one volume advection equation
512  alf_idx = adv_idx%end
513  sys_size = alf_idx !adv_idx%end
514 
515  IF (bubbles) THEN
516  bub_idx%beg = sys_size+1
517  bub_idx%end = sys_size+2*nb
518  IF (polytropic .NEQV. .true.) THEN
519  bub_idx%end = sys_size+4*nb
520  END IF
521  sys_size = bub_idx%end
522 
523  ALLOCATE( bub_idx%rs(nb), bub_idx%vs(nb) )
524  ALLOCATE( bub_idx%ps(nb), bub_idx%ms(nb) )
525  ALLOCATE( weight(nb),r0(nb),v0(nb) )
526 
527  DO i = 1, nb
528  IF (polytropic .NEQV. .true.) THEN
529  fac = 4
530  ELSE
531  fac = 2
532  END IF
533 
534  bub_idx%rs(i) = bub_idx%beg+(i-1)*fac
535  bub_idx%vs(i) = bub_idx%rs(i)+1
536 
537  IF (polytropic .NEQV. .true.) THEN
538  bub_idx%ps(i) = bub_idx%vs(i)+1
539  bub_idx%ms(i) = bub_idx%ps(i)+1
540  END IF
541  END DO
542 
543  IF (nb == 1) THEN
544  weight(:) = 1d0
545  r0(:) = 1d0
546  v0(:) = 0d0
547  ELSE IF (nb > 1) THEN
548  CALL s_simpson(nb)
549  v0(:) = 0d0
550  ELSE
551  stop 'Invalid value of nb'
552  END IF
553 
554  IF (polytropic .NEQV. .true.) THEN
555  CALL s_initialize_nonpoly
556  ELSE
557  rhoref = 1.d0
558  pref = 1.d0
559  END IF
560  END IF
561  END IF
562  ! ==================================================================
563 
564  ALLOCATE(mpi_io_data%view(1:sys_size))
565  ALLOCATE(mpi_io_data%var(1:sys_size))
566 
567  DO i = 1, sys_size
568  ALLOCATE(mpi_io_data%var(i)%sf(0:m,0:n,0:p))
569  mpi_io_data%var(i)%sf => null()
570  END DO
571 
572 
573  ! Size of the ghost zone layer is non-zero only when post-processing
574  ! the raw simulation data of a parallel multidimensional computation
575  ! in the Silo-HDF5 format. If this is the case, one must also verify
576  ! whether the raw simulation data is 2D or 3D. In the 2D case, size
577  ! of the z-coordinate direction ghost zone layer must be zeroed out.
578  IF(num_procs == 1 .OR. format /= 1 .OR. n == 0) THEN
579 
580  offset_x%beg = 0
581  offset_x%end = 0
582  offset_y%beg = 0
583  offset_y%end = 0
584  offset_z%beg = 0
585  offset_z%end = 0
586 
587  ELSEIF(p == 0) THEN
588 
589  offset_z%beg = 0
590  offset_z%end = 0
591 
592  END IF
593 
594 
595  ! Determining the finite-difference number and the buffer size. Note
596  ! that the size of the buffer is unrelated to the order of the WENO
597  ! scheme. Rather, it is directly dependent on maximum size of ghost
598  ! zone layers and possibly the order of the finite difference scheme
599  ! used for the computation of vorticity and/or numerical Schlieren
600  ! function.
601  buff_size = max( offset_x%beg, offset_x%end, offset_y%beg, &
602  offset_y%end, offset_z%beg, offset_z%end )
603 
604  IF(any(omega_wrt) .OR. schlieren_wrt) THEN
605  fd_number = max(1, fd_order/2)
606  buff_size = buff_size + fd_number
607  END IF
608 
609 
610  ! Allocating the grid variables in the x-coordinate direction
611  ALLOCATE(x_cb(-1-offset_x%beg : m + offset_x%end))
612  ALLOCATE(x_cc( -buff_size : m + buff_size ))
613  ALLOCATE( dx( -buff_size : m + buff_size ))
614 
615 
616  ! Allocating grid variables in the y- and z-coordinate directions
617  IF(n > 0) THEN
618 
619  ALLOCATE(y_cb(-1-offset_y%beg : n + offset_y%end))
620  ALLOCATE(y_cc( -buff_size : n + buff_size ))
621  ALLOCATE( dy( -buff_size : n + buff_size ))
622 
623  IF(p > 0) THEN
624  ALLOCATE(z_cb(-1-offset_z%beg : p + offset_z%end))
625  ALLOCATE(z_cc( -buff_size : p + buff_size ))
626  ALLOCATE( dz( -buff_size : p + buff_size ))
627  END IF
628 
629  ! Allocating the grid variables, only used for the 1D simulations,
630  ! and containing the defragmented computational domain grid data
631  ELSE
632 
633  ALLOCATE(x_root_cb(-1:m_root))
634  ALLOCATE(x_root_cc( 0:m_root))
635 
636  END IF
637 
638  IF (coarsen_silo) THEN
639  ALLOCATE(coarse_x_cb(-1-offset_x%beg : (m/2) + offset_x%end))
640  IF (n > 0) THEN
641  ALLOCATE(coarse_y_cb(-1-offset_y%beg : (n/2) + offset_y%end))
642  IF (p > 0) ALLOCATE(coarse_z_cb(-1-offset_z%beg : (p/2) + offset_z%end))
643  END IF
644  END IF
645 
646  IF (cyl_coord .NEQV. .true.) THEN ! Cartesian grid
647  grid_geometry = 1
648  ELSEIF (cyl_coord .AND. p == 0) THEN ! Axisymmetric cylindrical grid
649  grid_geometry = 2
650  ELSE ! Fully 3D cylindrical grid
651  grid_geometry = 3
652  END IF
653 
654  END SUBROUTINE s_initialize_global_parameters_module ! --------------------

657  SUBROUTINE s_initialize_nonpoly
658 
659  INTEGER :: ir
660  REAL(KIND(0.D0)) :: rhol0
661  REAL(KIND(0.D0)) :: pl0
662  REAL(KIND(0.D0)) :: uu
663  REAL(KIND(0.D0)) :: D_m
664  REAL(KIND(0.D0)) :: temp
665  REAL(KIND(0.D0)) :: omega_ref
666  REAL(KIND(0.D0)), DIMENSION(Nb) :: chi_vw0
667  REAL(KIND(0.D0)), DIMENSION(Nb) :: cp_m0
668  REAL(KIND(0.D0)), DIMENSION(Nb) :: k_m0
669  REAL(KIND(0.D0)), DIMENSION(Nb) :: rho_m0
670  REAL(KIND(0.D0)), DIMENSION(Nb) :: x_vw
671  ! polytropic index used to compute isothermal natural frequency
672  REAL(KIND(0.D0)), PARAMETER :: k_poly = 1.d0
673  ! universal gas constant
674  REAL(KIND(0.D0)), PARAMETER :: Ru = 8314.d0
675 
676 
677  ! liquid physical properties
678  REAL(KIND(0.D0)) :: mul0, ss, pv, gamma_v, M_v, mu_v
679 
680  ! gas physical properties
681  REAL(KIND(0.D0)) :: gamma_m, gamma_n, M_n, mu_n
682 
683  rhol0 = rhoref
684  pl0 = pref
685 
686  ALLOCATE( pb0(nb), mass_n0(nb), mass_v0(nb), pe_t(nb) )
687  ALLOCATE( k_n(nb), k_v(nb), omegan(nb) )
688  ALLOCATE( re_trans_t(nb), re_trans_c(nb), im_trans_t(nb), im_trans_c(nb) )
689 
690  pb0(:) = dflt_real
691  mass_n0(:) = dflt_real
692  mass_v0(:) = dflt_real
693  pe_t(:) = dflt_real
694  omegan(:) = dflt_real
695 
696  mul0 = fluid_pp(1)%mul0
697  ss = fluid_pp(1)%ss
698  pv = fluid_pp(1)%pv
699  gamma_v = fluid_pp(1)%gamma_v
700  m_v = fluid_pp(1)%M_v
701  mu_v = fluid_pp(1)%mu_v
702  k_v(:) = fluid_pp(1)%k_v
703 
704  gamma_n = fluid_pp(2)%gamma_v
705  m_n = fluid_pp(2)%M_v
706  mu_n = fluid_pp(2)%mu_v
707  k_n(:) = fluid_pp(2)%k_v
708 
709  gamma_m = gamma_n
710  IF (thermal==2 ) gamma_m = 1.d0 !isothermal
711 
712  temp = 293.15d0
713  d_m = 0.242d-4
714  uu = dsqrt( pl0/rhol0 )
715 
716  omega_ref = 3.d0*k_poly*ca + 2.d0*( 3.d0*k_poly-1.d0 )/web
717 
718  !!! thermal properties !!!
719  ! gas constants
720  r_n = ru/m_n
721  r_v = ru/m_v
722  ! phi_vn & phi_nv (phi_nn = phi_vv = 1)
723  phi_vn = ( 1.d0+dsqrt(mu_v/mu_n)*(m_n/m_v)**(0.25d0) )**2 &
724  / ( dsqrt(8.d0)*dsqrt(1.d0+m_v/m_n) )
725  phi_nv = ( 1.d0+dsqrt(mu_n/mu_v)*(m_v/m_n)**(0.25d0) )**2 &
726  / ( dsqrt(8.d0)*dsqrt(1.d0+m_n/m_v) )
727  ! internal bubble pressure
728  pb0 = pl0 + 2.d0*ss/( r0ref*r0 )
729 
730  ! mass fraction of vapor
731  chi_vw0 = 1.d0/( 1.d0+r_v/r_n*(pb0/pv-1.d0) )
732  ! specific heat for gas/vapor mixture
733  cp_m0 = chi_vw0*r_v*gamma_v/( gamma_v-1.d0 ) &
734  + ( 1.d0-chi_vw0 )*r_n*gamma_n/( gamma_n-1.d0 )
735  ! mole fraction of vapor
736  x_vw = m_n*chi_vw0/( m_v+(m_n-m_v)*chi_vw0 )
737  ! thermal conductivity for gas/vapor mixture
738  k_m0 = x_vw*k_v/( x_vw+(1.d0-x_vw)*phi_vn ) &
739  + ( 1.d0-x_vw )*k_n/( x_vw*phi_nv+1.d0-x_vw )
740  ! mixture density
741  rho_m0 = pv/( chi_vw0*r_v*temp )
742 
743  ! mass of gas/vapor computed using dimensional quantities
744  mass_n0 = 4.d0*( pb0-pv )*pi/( 3.d0*r_n*temp*rhol0 )*r0**3
745  mass_v0 = 4.d0*pv*pi/( 3.d0*r_v*temp*rhol0 )*r0**3
746  ! Peclet numbers
747  pe_t = rho_m0*cp_m0*uu*r0ref/k_m0
748  pe_c = uu*r0ref/d_m
749  ! nondimensional properties
750  r_n = rhol0*r_n*temp/pl0
751  r_v = rhol0*r_v*temp/pl0
752  k_n = k_n/k_m0
753  k_v = k_v/k_m0
754  pb0 = pb0/pl0
755  pv = pv/pl0
756 
757  ! bubble wall temperature, normalized by T0, in the liquid
758  ! keeps a constant (cold liquid assumption)
759  tw = 1.d0
760  ! natural frequencies
761  omegan = dsqrt( 3.d0*k_poly*ca+2.d0*(3.d0*k_poly-1.d0)/(web*r0) )/r0
762 
763  pl0 = 1.d0
764  DO ir = 1,nb
765  CALL s_transcoeff( omegan(ir)*r0(ir),pe_t(ir)*r0(ir), &
766  re_trans_t(ir),im_trans_t(ir) )
767  CALL s_transcoeff( omegan(ir)*r0(ir),pe_c*r0(ir), &
768  re_trans_c(ir),im_trans_c(ir) )
769  END DO
770  im_trans_t = 0d0
771  im_trans_c = 0d0
772 
773  rhoref = 1.d0
774  pref = 1.d0
775  END SUBROUTINE s_initialize_nonpoly

920  SUBROUTINE s_simpson( Npt )
921 
922  INTEGER, INTENT(IN) :: Npt
923  INTEGER :: ir
924  REAL(KIND(0.D0)) :: R0mn
925  REAL(KIND(0.D0)) :: R0mx
926  REAL(KIND(0.D0)) :: dphi
927  REAL(KIND(0.D0)) :: tmp
928  REAL(KIND(0.D0)) :: sd
929  REAL(KIND(0.D0)), DIMENSION(Npt) :: phi
930 
931  ! nondiml. min. & max. initial radii for numerical quadrature
932  !sd = 0.05D0
933  !R0mn = 0.75D0
934  !R0mx = 1.3D0
935 
936  sd = poly_sigma
937  r0mn = 0.8d0*dexp(-2.8d0 * sd)
938  r0mx = 0.2d0*dexp( 9.5d0 * sd) + 1.d0
939 
940  ! phi = ln( R0 ) & return R0
941  DO ir = 1,npt
942  phi(ir) = dlog( r0mn ) &
943  + dble( ir-1 )*dlog( r0mx/r0mn )/dble( npt-1 )
944  r0(ir) = dexp( phi(ir) )
945  END DO
946  dphi = phi(2) - phi(1)
947 
948  ! weights for quadrature using Simpson's rule
949  DO ir = 2,npt-1
950  ! Gaussian
951  tmp = dexp( -0.5d0*(phi(ir)/sd)**2 )/dsqrt( 2.d0*pi )/sd
952  IF ( mod(ir,2)==0 ) THEN
953  weight(ir) = tmp*4.d0*dphi/3.d0
954  ELSE
955  weight(ir) = tmp*2.d0*dphi/3.d0
956  END IF
957  END DO
958 
959  tmp = dexp( -0.5d0*(phi(1)/sd)**2 )/dsqrt( 2.d0*pi )/sd
960  weight(1) = tmp*dphi/3.d0
961  tmp = dexp( -0.5d0*(phi(npt)/sd)**2 )/dsqrt( 2.d0*pi )/sd
962  weight(npt) = tmp*dphi/3.d0
963 
964  END SUBROUTINE s_simpson
