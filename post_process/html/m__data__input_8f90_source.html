443 
444  ! Periodic BC at the beginning
445  ELSEIF(bc_x%beg == -1) THEN
446 
447  DO i = 1, buff_size

452  ELSE
453

457  x_cb(-1-i) = x_cb(-i) - dx(-i)

459

461  x_cc(-i) = x_cc(1-i) - (dx(1-i) + dx(-i))/2d0
462  END DO
463 
464  END IF

472

485  END DO
