478  SUBROUTINE s_initialize_global_parameters_module() ! -------------------
479 
480

488 
489  ! Determining the degree of the WENO polynomials
490  weno_polyn = (weno_order - 1) / 2
491 
492 
493  ! Initializing the number of fluids for which viscous effects will
494  ! be non-negligible, the number of distinctive material interfaces
495  ! for which surface tension will be important and also, the number
496  ! of fluids for which the physical and geometric curvatures of the
497  ! interfaces will be computed
498  re_size = 0; we_size = 0; crv_size = 0
499 
500 
501  ! Gamma/Pi_inf Model ===============================================
502  IF(model_eqns == 1) THEN
503 
504  ! Setting number of fluids
505  num_fluids = 1
506 
507  ! Annotating structure of the state and flux vectors belonging
508  ! to the system of equations defined by the selected number of
509  ! spatial dimensions and the gamma/pi_inf model
510  cont_idx%beg = 1
511  cont_idx%end = cont_idx%beg
512  mom_idx%beg = cont_idx%end + 1
513  mom_idx%end = cont_idx%end + num_dims
514  e_idx = mom_idx%end + 1
515  adv_idx%beg = e_idx + 1
516  adv_idx%end = adv_idx%beg + 1
517  gamma_idx = adv_idx%beg
518  pi_inf_idx = adv_idx%end
519  sys_size = adv_idx%end
520 
521  ! ==================================================================
522 
523 
524  ! Volume Fraction Model ============================================
525  ELSE
526 
527  ! Annotating structure of the state and flux vectors belonging
528  ! to the system of equations defined by the selected number of
529  ! spatial dimensions and the volume fraction model
530  IF(model_eqns == 2) THEN
531  cont_idx%beg = 1
532  cont_idx%end = num_fluids
533  mom_idx%beg = cont_idx%end + 1
534  mom_idx%end = cont_idx%end + num_dims
535  e_idx = mom_idx%end + 1
536  adv_idx%beg = e_idx + 1
537  adv_idx%end = e_idx + num_fluids
538 
539  IF( (adv_alphan .NEQV. .true.) .AND. &
540  (num_fluids > 1)) adv_idx%end = adv_idx%end - 1
541 
542  sys_size = adv_idx%end
543 
544  IF (bubbles) THEN
545  alf_idx = adv_idx%end
546  ELSE
547  alf_idx = 0
548  END IF
549 
550  IF (bubbles) THEN
551  bub_idx%beg = sys_size+1
552  bub_idx%end = sys_size+2*nb
553  IF (polytropic .NEQV. .true.) THEN
554  bub_idx%end = sys_size+4*nb
555  END IF
556  sys_size = bub_idx%end
557 
558  ALLOCATE( bub_idx%rs(nb), bub_idx%vs(nb) )
559  ALLOCATE( bub_idx%ps(nb), bub_idx%ms(nb) )
560  ALLOCATE( weight(nb),r0(nb),v0(nb) )
561 
562  DO i = 1, nb
563  IF (polytropic .NEQV. .true.) THEN
564  fac = 4
565  ELSE
566  fac = 2
567  END IF
568 
569  bub_idx%rs(i) = bub_idx%beg+(i-1)*fac
570  bub_idx%vs(i) = bub_idx%rs(i)+1
571 
572  IF (polytropic .NEQV. .true.) THEN
573  bub_idx%ps(i) = bub_idx%vs(i)+1
574  bub_idx%ms(i) = bub_idx%ps(i)+1
575  END IF
576  END DO
577 
578  IF (nb == 1) THEN
579  weight(:) = 1d0
580  r0(:) = 1d0
581  v0(:) = 0d0
582  ELSE IF (nb > 1) THEN
583  CALL s_simpson(nb)
584  v0(:) = 0d0
585  ELSE
586  stop 'Invalid value of nb'
587  END IF
588 
589  IF (polytropic .NEQV. .true.) THEN
590  CALL s_initialize_nonpoly
591  ELSE
592  rhoref = 1.d0
593  pref = 1.d0
594  END IF
595  END IF
596  ELSE IF(model_eqns == 3) THEN
597  cont_idx%beg = 1
598  cont_idx%end = num_fluids
599  mom_idx%beg = cont_idx%end + 1
600  mom_idx%end = cont_idx%end + num_dims
601  e_idx = mom_idx%end + 1
602  adv_idx%beg = e_idx + 1
603  adv_idx%end = e_idx + num_fluids
604  internalenergies_idx%beg = adv_idx%end + 1
605  internalenergies_idx%end = adv_idx%end + num_fluids
606  sys_size = internalenergies_idx%end
607  ELSE IF(model_eqns == 4) THEN
608  cont_idx%beg = 1 ! one continuity equation
609  cont_idx%end = 1 !num_fluids
610  mom_idx%beg = cont_idx%end + 1 ! one momentum equation in each direction
611  mom_idx%end = cont_idx%end + num_dims
612  e_idx = mom_idx%end + 1 ! one energy equation
613  adv_idx%beg = e_idx + 1
614  adv_idx%end = adv_idx%beg !one volume advection equation
615  alf_idx = adv_idx%end
616  sys_size = alf_idx !adv_idx%end
617 
618  IF (bubbles) THEN
619  bub_idx%beg = sys_size+1
620  bub_idx%end = sys_size+2*nb
621  IF (polytropic .NEQV. .true.) THEN
622  bub_idx%end = sys_size+4*nb
623  END IF
624  sys_size = bub_idx%end
625 
626 
627  ALLOCATE( bub_idx%rs(nb), bub_idx%vs(nb) )
628  ALLOCATE( bub_idx%ps(nb), bub_idx%ms(nb) )
629  ALLOCATE( weight(nb),r0(nb),v0(nb) )
630 
631  DO i = 1, nb
632  IF (polytropic .NEQV. .true.) THEN
633  fac = 4
634  ELSE
635  fac = 2
636  END IF
637 
638  bub_idx%rs(i) = bub_idx%beg+(i-1)*fac
639  bub_idx%vs(i) = bub_idx%rs(i)+1
640 
641  IF (polytropic .NEQV. .true.) THEN
642  bub_idx%ps(i) = bub_idx%vs(i)+1
643  bub_idx%ms(i) = bub_idx%ps(i)+1
644  END IF
645  END DO
646 
647  IF (nb == 1) THEN
648  weight(:) = 1d0
649  r0(:) = 1d0
650  v0(:) = 0d0
651  ELSE IF (nb > 1) THEN
652  CALL s_simpson(nb)
653  v0(:) = 0d0
654  ELSE
655  stop 'Invalid value of nb'
656  END IF
657 
658  IF (polytropic .NEQV. .true.) THEN
659  CALL s_initialize_nonpoly
660  ELSE
661  rhoref = 1.d0
662  pref = 1.d0
663  END IF
664  END IF
665  END IF
666 
667  ! Determining the number of fluids for which the shear and the
668  ! volume Reynolds numbers, e.g. viscous effects, are important
669  DO i = 1, num_fluids
670  IF(fluid_pp(i)%Re(1) > 0) re_size(1) = re_size(1) + 1
671  IF(fluid_pp(i)%Re(2) > 0) re_size(2) = re_size(2) + 1
672  END DO
673 
674  ! Determining the number of distinct material interfaces for
675  ! which the Weber number, e.g. the capillarity, is important
676  DO i = 1, num_fluids-1
677  DO j = i+1, num_fluids
678  IF(fluid_pp(i)%We(j) > 0) we_size = we_size + 1
679  END DO
680  END DO
681 
682  ! Constraining the utilization of WENO averaging so that it may
683  ! only be applied when the presence of viscous and/or capillary
684  ! effects requires that the cell-average spatial derivatives be
685  ! determined
686  IF( proc_rank == 0 .AND. weno_avg &
687  .AND. &
688  all(re_size == 0) .AND. we_size == 0 ) THEN
689  print '(A)', 'Unsupported combination of values ' // &
690  'of weno_avg, fluid_pp(:)%Re(:) and ' // &
691  'fluid_pp(:)%We(:). Exiting ...'
692  stop
693  END IF
694 
695  ! Bookkeeping the indexes of any viscous fluids and any pairs of
696  ! fluids whose interface will support effects of surface tension
697  IF(any(re_size > 0)) THEN
698 
699  ALLOCATE(re_idx(1:2,1:maxval(re_size)))
700 
701  k = 0
702  DO i = 1, num_fluids
703  IF(fluid_pp(i)%Re(1) > 0) THEN
704  k = k + 1; re_idx(1,k) = i
705  END IF
706  END DO
707 
708  k = 0
709  DO i = 1, num_fluids
710  IF(fluid_pp(i)%Re(2) > 0) THEN
711  k = k + 1; re_idx(2,k) = i
712  END IF
713  END DO
714 
715  END IF
716 
717  IF(we_size > 0) THEN
718 
719  ALLOCATE(we_idx(1:we_size,1:2))
720 
721  k = 0
722  DO i = 1, num_fluids-1
723  DO j = i+1, num_fluids
724  IF(fluid_pp(i)%We(j) > 0) THEN
725  k = k + 1; we_idx(k,:) = (/i,j/)
726  END IF
727  END DO
728  END DO
729 
730  crv_size = 2
731  DO i = 2, we_size
732  IF(we_idx(i,1) > we_idx(i-1,1)) THEN
733  crv_size = crv_size + 1
734  END IF
735  IF(all(we_idx(i,2) /= we_idx(1:i-1,2))) THEN
736  crv_size = crv_size + 1
737  END IF
738  END DO
739 
740  ALLOCATE(crv_idx(1:crv_size))
741 
742  k = 2; crv_idx(1:2) = we_idx(1,:)
743  DO i = 2, we_size
744  IF(we_idx(i,1) > we_idx(i-1,1)) THEN
745  k = k + 1; crv_idx(k) = we_idx(i,1)
746  END IF
747  IF(all(we_idx(i,2) /= we_idx(1:i-1,2))) THEN
748  k = k + 1; crv_idx(k) = we_idx(i,2)
749  END IF
750  END DO
751 
752  DO j = 1, crv_size
753  DO i = 1, crv_size-1
754  IF(crv_idx(i) > crv_idx(i+1)) THEN
755  tmp_idx = crv_idx(i+1)
756  crv_idx(i+1) = crv_idx( i )
757  crv_idx( i ) = tmp_idx
758  END IF
759  END DO
760  END DO
761 
762  END IF
763 
764  END IF
765  ! END: Volume Fraction Model =======================================
